name: Tests y Cobertura

on:
  push:
    branches: [ "main", "develop" ]
  pull_request:
    branches: [ "main", "develop" ]

jobs:
  unit-tests:
    name: Pruebas Unitarias
    runs-on: ubuntu-latest

    steps:
    - name: Checkout del repositorio
      uses: actions/checkout@v4

    - name: Configurar entorno de pruebas
      run: |
        echo "Configurando entorno de pruebas..."
        # Instale las dependencias necesarias aquí
        # Ejemplo para Node.js:
        # npm ci
        # Ejemplo para Python:
        # pip install -r requirements-dev.txt

    - name: Ejecutar pruebas unitarias
      run: |
        echo "Ejecutando pruebas unitarias..."
        # Ejemplo para Node.js:
        # npm run test:unit
        # Ejemplo para Python:
        # pytest tests/unit --cov=src --cov-report=xml

    - name: Subir reporte de cobertura
      uses: actions/upload-artifact@v4
      with:
        name: coverage-report
        path: ./coverage.xml
        retention-days: 14

  integration-tests:
    name: Pruebas de Integración
    runs-on: ubuntu-latest
    needs: unit-tests

    steps:
    - name: Checkout del repositorio
      uses: actions/checkout@v4

    - name: Ejecutar pruebas de integración
      run: |
        echo "Ejecutando pruebas de integración..."
        # Ejemplo:
        # npm run test:integration

  e2e-tests:
    name: Pruebas End-to-End
    runs-on: ubuntu-latest
    needs: integration-tests
    if: github.ref == 'refs/heads/develop' || github.ref == 'refs/heads/main'

    steps:
    - name: Checkout del repositorio
      uses: actions/checkout@v4

    - name: Ejecutar pruebas E2E
      run: |
        echo "Ejecutando pruebas end-to-end..."
        # Ejemplo con Playwright:
        # npx playwright test
